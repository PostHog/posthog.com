```js
// Capturing an event with groups
analytics.track('[event name]', {
    "$groups": {
        "company": "id:5"
    }
})

// Updating group properties
analytics.track('$groupidentify', {
    "$group_type": "company",
    "$group_key": "id:5",
    "$group_set": {
        "company_name": "Awesome Inc",
        "employees": 11
    }
})
```
