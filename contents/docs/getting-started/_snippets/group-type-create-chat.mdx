posthog.capture(
    'distinct_id',
    'Message sent',
    groups={'chat': '84359520'}
)

```node
posthog.capture({
    event: 'Message sent',
    distinctId: 'distinct_id',
    groups: { chat: '84359520' },
})

client.Enqueue(posthog.Capture{

    Event: "Message sent",
    Groups: posthog.NewGroups().
        Set("chat", "84359520").
})
```
